FROM FILM;

FROM FILM;

--2
SELECT *
FROM GLUMAC
WHERE BROJ = 38;

--3
--GRANICE INTERVALA NISU UKLJUCENE
SELECT BROJ, NASLOV
FROM FILM
WHERE GODINA > 1965 AND GODINA < 1975;

SELECT BROJ, NASLOV
FROM FILM
WHERE GODINA => 1965 AND GODINA <= 1975;

SELECT BROJ, NASLOV
FROM FILM
WHERE GODINA BETWEEN 1965 AND 1975;

--4
SELECT BROJ, NASLOV
FROM FILM
WHERE TIP = 'drama' OR TIP = 'comedy';

SELECT BROJ, NASLOV
FROM FILM
WHERE TIP IN ('drama', 'comedy');

--5
--POREDJENJE SA NULL VREDNOSTIMA NIJE MOGUCE
--ODNOSNO REZULTAT POREDJENJA JE UVEK NEDEFINISAN
SELECT IME, PREZIME, GODINA_ROD, MESTO_ROD
FROM GLUMAC
WHERE GODINA_SM = NULL;

SELECT IME, PREZIME, GODINA_ROD, MESTO_ROD
FROM GLUMAC
WHERE GODINA_SM IS NULL;

--6
SELECT IME, PREZIME, GODINA_SM - GODINA_ROD AS DUZINA_ZIVOTA
FROM REZISER
WHERE GODINA_SM IS NOT NULL;

--7
SELECT BROJ, NASLOV, AA_NOMINACIJE 
FROM FILM
WHERE AA_NOMINACIJE IS NOT NULL OR AA_NOMINACIJE > 0;

--8
SELECT SUM(AA_NAGRADE)
FROM FILM
WHERE TIP = 'comedy';

SELECT SUM(AA_NAGRADE)
FROM FILM
WHERE TIP = 'comedy'  AND (AA_NAGRADE IS NOT NULL AND AA_NAGRADE > 0);

--9

FROM IGRA

--10

SELECT BROJ, NASLOV
FROM FILM
WHERE NASLOV LIKE 'bird';

SELECT BROJ, NASLOV
FROM FILM
WHERE NASLOV LIKE '%bird';

SELECT BROJ, NASLOV
FROM FILM
WHERE NASLOV LIKE 'bird%';

FROM FILM
WHERE NASLOV LIKE '%bird%';

FROM FILM
