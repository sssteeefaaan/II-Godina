SELECT AUTOR.IME, AUTOR.PREZIME,COUNT(*) AS Broj_knjiga
FROM AUTOR, JE_NAPISAO
WHERE AUTOR.ID=JE_NAPISAO.ID
GROUP BY(AUTOR.ID,AUTOR.IME,AUTOR.PREZIME)
HAVING COUNT(*) =
			(
				SELECT MAX(COUNT(*))
				FROM AUTOR,JE_NAPISAO
				WHERE JE_NAPISAO.ID=AUTOR.ID
				GROUP BY(JE_NAPISAO.ID)
			);

SELECT U.Naziv_discipline,UK.Ime, UK.Prezime, EXTRACT(YEAR FROM(SYSDATE))-EXTRACT(YEAR FROM(UK.Datum_rodjenja)) AS Starost
FROM UCESNIK UK,UCESTVUJE U
WHERE UK.Sifra=U.Sifra AND
EXTRACT(YEAR FROM(SYSDATE))-EXTRACT(YEAR FROM(UK.Datum_rodjenja))
> 	(
		SELECT AVG(EXTRACT(YEAR FROM(SYSDATE))-EXTRACT(YEAR FROM(UK.Datum_rodjenja)))
		FROM UCESNIK UK
		WHERE UK.Sifra=U.Sifra
		GROUP BY(U.Naziv_discipline)
	);
